import { useState } from 'react'

const App = () => {
   var [a, setA] = useState("A") // a 는 읽기 전용변수로 useState 소괄호 안의 값을 담는 변수라 한다. 그래서 초기값이 "A"가 된다.
   let [b, setB] = useState("B")
   const [c, setC] = useState("C")  
   const fn1 = () => console.log("fn1")
   const fn2 = () => console.log("fn2")
   // 이걸 쓰면 배열값이 전달된다. 자바스크립트에서는 변수로 받는다. const만 usestate로 받을때만 제약이 걸리므로 리엑트 스테이트를 쓸수 있다. let var는 쓰면 안된다.
   // 앞에 있는 변수가 제약이 걸려야 뒤에 있는 함수를 사용할 이유가 있다.
  console.log("앱 호출 됨")
  const event = () => {
    console.log(1)
    if (a === "A") return
    fn1()
    console.log(2)
    fn2()
    console.log(3)
  } //순서는 무조건 useState에 설정된 함수값이 나온 뒤에 app함수로 가서 실행된다.
//useState 쓸때 함수 안에 setA와 같이 if문을 사용하면 아무런 의미가 없다. 왜냐하면 setA의 값이 함수 끝나고 바뀌기 때문에 내부에 확인하는 if문은 아무쓸모가 없다. 그래서 함수 밖에서 위에 if문을 사용하자!

  return (
    <>
      <button type="button" onClick={event}>호출</button>
      <p>{a}</p>
      <p>{b}</p>
      <p>{c}</p>
    </>
  )
}

export default App
